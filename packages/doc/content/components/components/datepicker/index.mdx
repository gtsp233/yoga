# Datepicker

### Reference

The `<Datepicker />` component is a date picker that allows the user to select a single date or a range of dates.

Both `startDate` and `endDate` are expected to be in the [`Date`](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Date) format.

### Usage

### Single date Datepicker

```javascript state
render(() => {
  const [startDate, setStartDate] = useState();
  const [endDate, setEndDate] = useState();
  const [selectedDate, setSelectedDate] = useState();

  const onSelFunc = date => {
    setSelectedDate(date);
    console.log('some Data =>', date);
  };

  const onSelectRange = (startDate, endDate) => {
    console.log({ startDate, endDate });
  };

  return (
    <Datepicker
      startDate={startDate}
      endDate={endDate}
      isMonthDisplay
      type="range"
      onSelectRange={onSelectRange}
    />
  );
});
```

### Single date Datepicker

### Range date Datepicker

```javascript state
render(() => {
  const [startDate, setStartDate] = useState();
  const [endDate, setEndDate] = useState();

  const onSelectRange = (startDate, endDate) => {
    console.log({ startDate, endDate });
  };

  return (
    <Datepicker
      type="range"
      startDate={startDate}
      endDate={endDate}
      onSelectRange={onSelectRange}
    />
  );
});
```

### Range date Datepicker with custom onSelectRange

```javascript state
render(() => {
  const [startDate, setStartDate] = useState();
  const [endDate, setEndDate] = useState();

  const onCustomSelectRange = selectedDate => {
    console.log({ selectedDate });
    if (!startDate || (startDate > selectedDate && !endDate) || endDate) {
      setStartDate(selectedDate);
      if (endDate) setEndDate(undefined);
    } else {
      setEndDate(selectedDate);
    }
  };

  return (
    <Datepicker
      type="range"
      startDate={startDate}
      endDate={endDate}
      customOnSelectRange={onCustomSelectRange}
    />
  );
});
```
